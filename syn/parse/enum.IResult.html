extern crate syn;

use syn::parse::ty;

// One or more Rust types separated by commas.

    separated_nonempty_list!(punct!(","), ty)
);

fn main() {
    let input = "&str, Map<K, V>, String";

    let parsed = comma_separated_types(input).expect("comma-separated types");

    assert_eq!(parsed.len(), 3);
    println!("{:?}", parsed);
